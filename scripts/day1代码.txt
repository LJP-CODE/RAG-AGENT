import json
import time
from pathlib import Path


base_dir = Path(__file__).resolve().parent
input_dir = base_dir / "day_1data" / "day1_data" / "input"
output_dir = base_dir / "day_1data" / "day1_data" / "output"
output_json = output_dir / "translation_report.json"

output_dir.mkdir(parents=True, exist_ok=True)
files = sorted(input_dir.glob("*.txt"))

results = []
failures = []


def translate_to_english(text: str) -> str:
    if not text.strip():
        return ""

    if all(ord(ch) < 128 for ch in text):
        return text

    try:
        from deep_translator import GoogleTranslator

        translator = GoogleTranslator(source="auto", target="en")
        return translator.translate(text)
    except Exception:
        return text


start_total = time.time()
for file_path in files:
    start = time.time()
    try:
        with file_path.open("r", encoding="utf-8") as handle:
            content = handle.read()

        translated_text = translate_to_english(content)
        output_path = output_dir / file_path.name
        with output_path.open("w", encoding="utf-8") as handle:
            handle.write(translated_text)

        elapsed = time.time() - start
        results.append(
            {
                "file": file_path.name,
                "status": "success",
                "response_time_seconds": round(elapsed, 6),
                "output_file": output_path.name,
                "input_bytes": len(content.encode("utf-8")),
                "output_bytes": len(translated_text.encode("utf-8")),
            }
        )
    except Exception as exc:
        elapsed = time.time() - start
        failures.append(
            {
                "file": file_path.name,
                "status": "failure",
                "response_time_seconds": round(elapsed, 6),
                "error": str(exc),
            }
        )

report = {
    "input_dir": str(input_dir),
    "output_dir": str(output_dir),
    "total_files": len(files),
    "success_count": len(results),
    "failure_count": len(failures),
    "total_elapsed_seconds": round(time.time() - start_total, 6),
    "success_files": results,
    "failed_files": failures,
}

with output_json.open("w", encoding="utf-8") as handle:
    json.dump(report, handle, ensure_ascii=False, indent=2)

print("统计报告:")
print(json.dumps(report, ensure_ascii=False, indent=2))
print(f"\nJSON report saved to: {output_json}")
